\par 有限维空间 $V$ 的\link{线性空间}{Hamel基}简称为 $V$ 的\newconcept{基}, 我们将会看到, \link{}{}(基于拓扑结构定义的基)与Hamel基\link{}{}(在有限维空间中等价).

\Theorem{维数}{
    有限维空间的任何基的索引集都是有限集且大小相同. 设 $(\bm e_{k})_{k=1}^{n}$ 是 $V$ 的基, 称 $V$ 为 \newconcept{$n$ 维线性空间}并记其维数为 $\dim V=n$ .
}
\Proof{证明}{
    \par 设 $(\bm e_{k}^{(1)})_{k=1}^{n_{1}}$ 和 $(\bm e_{k}^{(2)})_{k=1}^{n_{2}}$ 都是 $V$ 的基, 注意 $(\bm e_{k}^{(1)})_{k=1}^{n_{1}}$ 线性无关且 $(\bm e_{k}^{(2)})_{k=1}^{n_{2}}$ 张成 $V$ , 由\link{线性空间}[向量组]{替换定理}知 $n_{1}\leqslant n_{2}$ , 反过来又有 $n_{1}\geqslant n_{2}$ 成立, 于是 $n_{1}=n_{2}$ .\ProofEndT
}

\par 对于有限向量组 $S$ , 我们称 $\dim\spn S$ 为 $S$ 的\newconcept{秩}.

\par 需要注意, 线性空间 $V$ 与其所在域 $\bfF$ 有密切关系, 例如, 实线性空间 $\bfR^{2}$ 和复线性空间 $\bfC$ 这样的集合的维数是不同的, 即使它们在集合角度是完全同构的.

\par 就直觉而言, 我们似乎可以通过对无关组/张成组进行缩减/扩充以得到线性空间的基, 下面的两则定理表明我们确实可以做到这一点.

\Theorem{基的缩减构造}{
    若向量组 $(\bm e_{k})_{k\in I}$ 张成 $V$ , 则存在 $J\subseteq I$ 使得 $(\bm e_{k})_{k\in J}$ 是 $V$ 的基.
}
\Proof{证明}{
    \par 从 $k=1$ 开始, 依次考虑 $S$ 中的每个向量 $\bm v_{k}$ , 当且仅当 $\bm v_{k}\notin\spn\braIII{\cdots\and\bm v_{k-1}}$ 时将 $\bm v_{k}$ 保留于 $B$ . 从而我们得到了一个新的向量组 $B$ , 其与 $S$ 等价因此满足 $\spn B=V$ , 并且 $B$ 中任何一个向量都不能被表示为其之前的其它向量的线性组合, 即 $B$ 线性无关, 故 $B$ 是 $V$ 的Hamel基.\ProofEndT
}

\Theorem{基的扩充构造}{
    若向量组 $(\bm e_{k})_{k=1}^{n}$ 线性无关, 则存在 $J\supseteq I$ 使得 $(\bm e'_{k})_{k\in J}$ 是 $V$ 的基并且 $\bm e_{k}=\bm e'_{k}$ 对一切 $k\in I$ 成立.
}
\Proof{证明}{
    \par 因为 $V$ 是一个有限维空间, 所以可以找到某个向量组 $S'$ 使得 $\spn S'=V$ , 考虑对向量组 $S\cup S'$ 应用\link{维数}{基的缩减构造}的构造过程即可得 $V$ 的Hamel基 $B$ , 并且 $S\subseteq B$ , 这是因为 $S$ 线性无关, 其中的任何向量 $\bm v_{k}$ 都无法表示为其之前的向量的线性组合, 因此必定会被保留于 $B$ .\ProofEndT
}

\Theorem{基的维数准则}{
    若向量组 $(\bm e_{k})_{k=1}^{\dim\!V}$ 线性无关或张成 $V$ , 则 $(\bm e_{k})_{k=1}^{\dim\!V}$ 是 $V$ 的基.
}
\Proof{证明}{
    \par 当 $(\bm e_{k})_{k=1}^{\dim\!V}$ 张成 $V$ 时, 由\link{有限维线性空间}{基的缩减构造}可知存在 $(\bm e_{k})_{k=1}^{\dim\!V}$ 的子组 $(\bm e'_{k})_{k=1}^{n}$ 是 $V$ 的基, 又 $\dim V=n$ , 故必有 $(\bm e_{k})_{k=1}^{\dim\!V}=(\bm e'_{k})_{k=1}^{n}$ .
    \par 当 $(\bm e_{k})_{k=1}^{\dim\!V}$ 线性无关时, 由\link{有限维线性空间}{基的扩充构造}可知存在 $(\bm e_{k})_{k=\dim\!V+1}^{n}$ 使得 $(\bm e_{k})_{k=1}^{n}$ 是 $V$ 的基, 又 $\dim V=n$ , 故必有 $(\bm e_{k})_{k=1}^{\dim\!V}=(\bm e'_{k})_{k=1}^{n}$ .\ProofEndT
}

\Theorem{子空间的维数}{
    设 $U$ 是 $V$ 的子空间, 则 $$\dim U\leqslant\dim V\ ,$$ 其中等号当且仅当 $U=V$ 时取得.
}
\Proof{证明}{
    \par 将 $U$ 的基看作 $V$ 中线性无关的向量组, 将 $V$ 的基看作 $V$ 中张成 $V$ 的向量组, 由\link{线性空间}[向量组]{替换定理}即得该不等式.
    \par 现在考虑取等号的情形. 一方面, 如果 $U=V$ , 则显然 $\dim U=\dim V$ ; 另一方面, 如果 $\dim U=\dim V$ , 设 $B$ 是 $U$ 的基, 则由\link{有限维线性空间}{基的维数准则}可知 $\spn B=V$ , 故 $U=V$ .\ProofEndT
}
