\Definition{对称多重线性型}{
    \,$m$ 重线性型 $\rho$ 是\newconcept{对称型}, 当且仅当其满足:
    \serial{•}对称性:
    \begin{align*}
        \rho\braI{\cdots\and\underset{\text{第$p$个}}{\bm u}\and\cdots\and\underset{\text{第$q$个}}{\bm v}\and\cdots}=\rho\braI{\cdots\and\underset{\text{第$p$个}}{\bm v}\and\cdots\and\underset{\text{第$q$个}}{\bm u}\and\cdots}
    \end{align*}
    对一切 $\bm u\and\bm v\in V$ 和正整数 $p\leqslant q\leqslant m$ 成立.
    \par\noindent 全体 $V$ 上的对称 $m$ 重线性型构成的集合记为 $\mathcal{L}_{\mathrm{sym}}^{m}\braI{V\by\bfF}$ .
}

\vspace{-4pt}

\Theorem{对称多重线性型构成线性空间}{
    \serial{1.}$n=0$ 时: $\mathcal{L}_{\mathrm{sym}}^{m}\braI{V\by\bfF}=\braIII{\bm 0}$ .
    \serial{2.}$n\ne 0$ 时: $\braI{\mathcal{L}_{\mathrm{sym}}^{m}\braI{V\by\bfF}\and\bfF\and+\and\cdot}$ 是 $\mathcal{L}^{m}\braI{V\by\bfF}$ 的 $\rmC_{n+m-1}^{m}$ 维子空间.
}
